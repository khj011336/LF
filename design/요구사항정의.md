### DB 스키마 (ERD - 물리적) 

#### members
- int id <<PK>> 회원번호 (개념) AI(AutoIncrement)
- varchar(12) name 이름 NN(NotNull)
- varchar(12) login <<UQ>> 로그인 NN
- varchar(12) pw 암호 NN
- timestamp joinedAt 가입일 CURRENT_TIMESTAMP

#### blogs
- int id <<PK>> 게시글번호 AI
- varchar(64) title 제목 
- varchar(4096) content 내용
- int readCount 조회수 '0'
- int likeCount 좋아요수 '0'
- String likeMembers 좋아요한 회원들(id를 csv로 나열) 
		"1,30,21" => 3명의 좋아요 
- timestamp createdAt 등록일	CURRENT_TIMESTAMP	 
- timestamp updatedAt 최근갱신일 CURRENT_TIMESTAMP
- int memberId <<FK>> 게시글 작성자의 회원번호 NN

#### replies
- int id <<PK>> 댓글번호 AI
- varchar(145) memo 내용 NN
- timestamp created_at 등록일	CURRENT_TIMESTAMP	 
- int member_id <<FK>> 댓글 작성자의 회원번호 NN
- int blog_id <<FK>> 댓글 대상 게시글의 고유번호 NN
